**Gawin mo ang mga method file ng `index.ipc.js`.**  
Siguraduhin na:  
- **Robust** ang bawat method file (may malinaw na error handling at standardized response format).  
- **Type‑safe** gamit ang JSDoc o TypeScript annotations para sa parameters at return values.  
- **Error‑safe** (lahat ng IPC calls may `try/catch` at consistent `{ status, message, data }` response).  
- **Consistent logging** (mag‑print ng `[IPC] <method>` sa bawat call para madaling i‑trace).  
- **Import clarity** (i‑print lahat ng method files na naka‑import sa `index.ipc.js`).  

---

### Example: `index.ipc.js`
```js
// src/main/ipc/index.ipc.js
//@ts-check

const addPoints = require("./loyalty/add_points.ipc");
const createLoyalty = require("./loyalty/create.ipc");
const refundSale = require("./sale/refund.ipc");
const createSale = require("./sale/create.ipc");
const updateProduct = require("./product/update.ipc");
const createProduct = require("./product/create.ipc");

console.log("[IPC] Loaded methods:", {
  loyalty: ["add_points", "create"],
  sale: ["refund", "create"],
  product: ["update", "create"],
});

module.exports = {
  loyalty: {
    addPoints,
    create: createLoyalty,
  },
  sale: {
    refund: refundSale,
    create: createSale,
  },
  product: {
    update: updateProduct,
    create: createProduct,
  },
};
```

### Example Method File (robust + type‑safe)
```js
// src/main/ipc/loyalty/add_points.ipc.js
//@ts-check
const auditLogger = require("../../../utils/auditLogger");

/**
 * Add loyalty points to a customer
 * @param {Object} params
 * @param {number} params.customerId
 * @param {number} params.points - Positive number of points to add
 * @param {string} [params.notes]
 * @param {number} [params.saleId]
 * @param {string} [params.user]
 * @param {import("typeorm").QueryRunner} queryRunner
 * @returns {Promise<{status: boolean, message: string, data: any}>}
 */
module.exports = async (params, queryRunner) => {
  try {
    if (!params.customerId) {
      return { status: false, message: "customerId is required", data: null };
    }
    if (!params.points || params.points <= 0) {
      return { status: false, message: "points must be a positive number", data: null };
    }

    const createHandler = require("./create.ipc");
    const result = await createHandler(
      {
        customerId: params.customerId,
        pointsChange: params.points,
        notes: params.notes || "Points added",
        saleId: params.saleId,
        user: params.user,
      },
      queryRunner
    );

    console.log("[IPC] loyalty:add_points success", result);
    return result;
  } catch (error) {
    console.error("[IPC] loyalty:add_points error", error);
    return {
      status: false,
      message: error.message || "Failed to add loyalty points",
      data: null,
    };
  }
};
```

### ✅ Benefits
- **Type safety** → JSDoc annotations make payloads and return values explicit.  
- **Error safety** → standardized `{ status, message, data }` responses.  
- **Robust logging** → every IPC method prints its lifecycle.  
- **Centralized imports** → `index.ipc.js` clearly shows all available methods.
